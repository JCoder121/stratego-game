# Stratego Web App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Web app for the existing Stratego engine: online human-vs-human via room codes, human-vs-bot, and bot-vs-bot watch mode, served by one local Node server.

**Architecture:** Single Node process â€” hand-rolled HTTP static server + `ws` WebSocket endpoint. Server is authoritative: it owns `GameState` per room and only ever sends redacted `PlayerView`s (or an all-revealed `WatchView` to spectators / at game end). Client is a framework-free Vite TS app that renders the latest view and sends actions. Setup is staged entirely client-side and committed atomically (`COMMIT_SETUP`), because the engine has no un-place/swap action.

**Tech Stack:** TypeScript ESM (NodeNext, `.js` import suffixes), vitest, `ws` (only new runtime dep), Vite + vanilla TS client, `@playwright/test` for the final smoke test.

## Global Constraints

- `src/engine`, `src/bots`, `src/cli`, `src/sim`, `src/rng` are **untouched**. All 107 existing tests stay green (`npm test`).
- Enemy ranks and real enemy `PieceId`s (they encode rank: `RED-MARSHAL-0`) must never appear in any message to a player pre-reveal. Wire messages use squares + public ranks only â€” never forward raw `GameEvent`s (STRIKE events contain real ids).
- Strict TS, `tsc --noEmit` clean. Match existing style: named exports, small files, no classes in the engine â€” server may use classes for stateful Room objects.
- New runtime dep: `ws` only. Dev deps added: `@types/ws`, `vite`, `@playwright/test`.
- All server tests run offline/in-process with injected clock/scheduler/transport. Only Task 6's integration suite and Task 12's e2e open real sockets (loopback, ephemeral ports) â€” like the existing sim/E2E suites they may need the sandbox disabled; run them via their env-gated scripts.
- Server binds `0.0.0.0`, port `process.env.PORT ?? 3000`.
- Commit after every task (steps include commits).

```ts
// src/server/protocol.ts
import type { Action, Color, GameResult, Phase, PieceId, Rank, Square } from '../engine/index.js';
import type { PlayerView } from '../engine/index.js';

export type Mode = 'HUMAN_VS_HUMAN' | 'HUMAN_VS_BOT' | 'BOT_VS_BOT';
export type BotKind = 'random' | 'heuristic';
export type WatchSpeed = 500 | 1000 | 'step';
export type Role = Color | 'SPECTATOR';

/** All-revealed view for spectators and game-over broadcast. */
export interface WatchView {
  phase: Phase;
  turn: Color;
  plyCount: number;
  pieces: { owner: Color; pos: Square; rank: Rank; revealed: boolean }[];
  result: GameResult | null;
}

export interface StrikeSummary {
  attackerRank: Rank;
  defenderRank: Rank;
  outcome: 'ATTACKER' | 'DEFENDER' | 'BOTH';
}
export interface LastMove { from: Square; to: Square; by: Color; strike?: StrikeSummary }

/** Ranks are public once captured (every capture goes through a rank-revealing strike). */
export type CapturedRanks = Record<Color, Rank[]>;

export type ClientMsg =
  | { t: 'CREATE_ROOM'; mode: Mode; botDifficulty?: BotKind; bots?: { RED: BotKind; BLUE: BotKind }; watchSpeed?: WatchSpeed }
  | { t: 'JOIN_ROOM'; code: string }
  | { t: 'REJOIN'; code: string; token: string }
  | { t: 'COMMIT_SETUP'; placement: [PieceId, Square][] }
  | { t: 'ACTION'; action: Action; seq: number }        // MOVE / RESIGN only; setup goes via COMMIT_SETUP
  | { t: 'REMATCH_REQUEST' }
  | { t: 'WATCH_CONTROL'; control: 'play' | 'pause' | 'step' | 'speed'; speed?: WatchSpeed };

export type ServerMsg =
  | { t: 'ROOM_CREATED'; code: string; token: string; role: Role }
  | { t: 'JOINED'; code: string; token: string; role: Role }
  | { t: 'VIEW'; view: PlayerView | WatchView; captured: CapturedRanks; lastMove?: LastMove; seq: number }
  | { t: 'SETUP_STATUS'; ready: Record<Color, boolean> }
  | { t: 'GAME_OVER'; result: GameResult; finalView: WatchView; captured: CapturedRanks }
  | { t: 'OPPONENT_STATUS'; seat: Color; connected: boolean }
  | { t: 'REMATCH_STATE'; votes: Role[] }
  | { t: 'ERROR'; code: 'BAD_MSG' | 'NO_ROOM' | 'ROOM_FULL' | 'BAD_TOKEN' | 'NOT_YOUR_TURN' | 'INVALID_ACTION' | 'BAD_SETUP'; msg: string };

const CLIENT_TYPES = new Set(['CREATE_ROOM', 'JOIN_ROOM', 'REJOIN', 'COMMIT_SETUP', 'ACTION', 'REMATCH_REQUEST', 'WATCH_CONTROL']);

export function isClientMsg(x: unknown): x is ClientMsg {
  if (typeof x !== 'object' || x === null) return false;
  const m = x as Record<string, unknown>;
  if (typeof m.t !== 'string' || !CLIENT_TYPES.has(m.t)) return false;
  switch (m.t) {
    case 'JOIN_ROOM': return typeof m.code === 'string';
    case 'REJOIN': return typeof m.code === 'string' && typeof m.token === 'string';
    case 'COMMIT_SETUP': return Array.isArray(m.placement);
    case 'ACTION': return typeof m.action === 'object' && m.action !== null && typeof m.seq === 'number';
    case 'CREATE_ROOM': return typeof m.mode === 'string';
    default: return true;
  }
}
```

```ts
export class GameRoom {
  private state: GameState;
  private members = new Map<string, { role: Role; connected: boolean; send: (m: ServerMsg) => void }>();
  private botSeats: Partial<Record<Color, Bot>> = {};
  private rng: Rng;                       // makeSeeded(opts.seed) if given else makeRandom()
  // ...
  private commitSetup(color: Color, placement: [PieceId, Square][]) {
    let scratch = this.state;
    for (const [pieceId, to] of placement) {
      const action: Action = { type: 'SETUP_PLACE', color, pieceId, to };
      const err = validateAction(scratch, action);
      if (err) return this.sendTo(color, { t: 'ERROR', code: 'BAD_SETUP', msg: err });
      scratch = strategoReduce(scratch, action).state;
    }
    const done: Action = { type: 'SETUP_DONE', color };
    const err = validateAction(scratch, done);
    if (err) return this.sendTo(color, { t: 'ERROR', code: 'BAD_SETUP', msg: err });
    this.state = strategoReduce(scratch, done).state;   // adopt atomically
    this.broadcastSetupStatus();
    if (this.state.phase === 'PLAY') this.broadcastViews(undefined);
  }
  private viewMsg(role: Role, lastMove?: LastMove): ServerMsg {
    const view = role === 'SPECTATOR' ? watchView(this.state) : viewFor(this.state, role);
    return { t: 'VIEW', view, captured: capturedRanks(this.state), lastMove, seq: this.seq };
  }
}

  const out: CapturedRanks = { RED: [], BLUE: [] };
  for (const p of Object.values(state.pieces)) if (p.pos === null) out[p.owner].push(p.rank);
  return out;
}
```
